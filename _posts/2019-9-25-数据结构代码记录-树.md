



## 遍历



### 中序遍历



- 递归

  ```c++
  /**
   * Definition for a binary tree node.
   * struct TreeNode {
   *     int val;
   *     TreeNode *left;
   *     TreeNode *right;
   *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
   * };
   */
  class Solution {
      
  private:
      vector<int> ans;
      
  public:
      vector<int> inorderTraversal(TreeNode* root) {
          
          if(root) {
              inorderTraversal(root->left);
              ans.push_back(root->val);
              inorderTraversal(root->right);
          }
          
          return ans;
      }
  };
  ```

  



- 迭代

  ```C++
  class Solution {
      
  public:
      vector<int> inorderTraversal(TreeNode* root) {
          
          vector<int> ans;
          stack<TreeNode> s;
          TreeNode* t = root;
          
          while(t != nullptr || s.empty() == false) {
              
              while(t != nullptr) {
                  s.push(*t);
                  t = t->left;
              }
              
              if(s.empty() == false) {
                  ans.push_back(s.top().val);
                  t = s.top().right;
                  s.pop();
              }
          }
          
          return ans;
      }
  };
  ```
